ans 50: SELECT emp_name, salary*12 FROM Employee WHERE salary*12 > ALL(SELECT salary*12 FROM Employee WHERE dept_name='IT');

	select e1.emp_name, (e1.salary*12) as annual_salary from employee e1 
	join (select max(salary) as max_salary 
	from employee 
	where dept_name='IT') as e2 
	ON (e1.salary*12) > (e2.max_salary*12);

	select e.emp_name, (e.salary*12) from employee e, employee s 
	where s.dept_name='IT' 
	group by e.emp_name, e.salary 
	having (e.salary*12) > MAX(s.salary*12);
