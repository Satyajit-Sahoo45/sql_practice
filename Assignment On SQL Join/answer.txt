EASY LEVEL: 
----------

1. SELECT E.ENAME, D.DNAME
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO;

2. SELECT E.JOB, D.LOC
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO;

3. SELECT E.ENAME, E.SAL, D.DNAME 
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO;

4. SELECT E.ENAME, D.DNAME
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO
   WHERE D.DNAME='SALES';

5.SELECT E.ENAME, D.DNAME
  FROM EMP E
  JOIN DEPT D
  ON E.DEPTNO=D.DEPTNO
  WHERE D.LOC='DALLAS';

6. SELECT E.ENAME, E.JOB, D.DNAME
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO
   WHERE E.JOB='MANAGER';

7.SELECT E.ENAME, D.DEPTNO
  FROM EMP E
  JOIN DEPT D
  ON E.DEPTNO=D.DEPTNO;

8. SELECT E.ENAME, E.HIREDATE, D.LOC
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO;

9. SELECT E.ENAME, D.DNAME
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO
   WHERE D.DNAME='ACCOUNTING';

10. SELECT E.ENAME, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    ORDER BY D.DNAME;

11. SELECT E.*, D.* 
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    WHERE D.DEPTNO=20;

12. SELECT E.ENAME, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    WHERE D.LOC='CHICAGO';

13. SELECT E.*, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    WHERE E.SAL>2000;

14. SELECT E.ENAME, D.DNAME, D.LOC
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO;

15. SELECT E.ENAME, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    ORDER BY E.ENAME;



INTERNEDIATE LEVEL:
------------------
1. SELECT D.*, E.*
   FROM DEPT D
   LEFT JOIN EMP E
   ON D.DEPTNO=E.DEPTNO;

2. SELECT E.ENAME, D.DNAME
   FROM EMP E
   LEFT JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO;

   SELECT D.DNAME, E.ENAME
   FROM DEPT D
   LEFT JOIN EMP E
   ON D.DEPTNO=E.DEPTNO;

3. SELECT E.ENAME, D.DNAME
   FROM EMP E
   RIGHT JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO;

4. SELECT E.ENAME 
   FROM EMP E
   LEFT JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO
   WHERE D.DEPTNO IS NULL;

5. SELECT D.DNAME
   FROM DEPT D
   LEFT JOIN EMP E
   ON D.DEPTNO=E.DEPTNO
   WHERE E.EMPNO IS NULL;

6. SELECT E.ENAME, D.DNAME, D.LOC
   FROM EMP E
   LEFT JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO;

7. SELECT D.DNAME, COUNT(*) AS EMP_COUNT
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO
   GROUP BY D.DNAME;

8. SELECT E.ENAME, D.LOC
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO
   WHERE D.LOC IN ('NEW YORK', 'CHICAGO');

9. SELECT D.DNAME, MAX(E.SAL) AS MAX_SAL
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO=D.DEPTNO
   GROUP BY D.DNAME;

10. SELECT E.ENAME, E.SAL, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    WHERE E.SAL > (SELECT AVG(SAL) FROM EMP WHERE DEPTNO = E.DEPTNO);

11. SELECT E.ENAME, D.DNAME 
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    WHERE E.COMM > 0 AND E.COMM IS NOT NULL;

12. SELECT E.ENAME, M.ENAME AS MANAGER
    FROM EMP E
    JOIN EMP M
    ON E.MGR = M.EMPNO
    WHERE E.HIREDATE > M.HIREDATE;

13. SELECT E.ENAME, M.ENAME, D.DNAME
    FROM EMP E
    JOIN EMP M
    ON E.MGR = M.EMPNO
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO;

14. SELECT E.ENAME, D.DNAME AS EMP_DEPT_NAME, M.ENAME AS MANAGER, DD.DNAME AS MANAGER_DEPT_NAME
    FROM EMP E
    JOIN EMP M
    ON E.MGR = M.EMPNO
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    JOIN DEPT DD
    ON M.DEPTNO = DD.DEPTNO;

15. SELECT D.DNAME, AVG(E.SAL) AS DEPT_AVG_SAL
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    GROUP BY D.DNAME
    HAVING AVG(E.SAL) > 2000;



ADVANCED LEVEL:
--------------
1. SELECT E.ENAME, D.DNAME AS EMP_DEPT_NAME, M.ENAME AS MANAGER_NAME, DD.DNAME AS MANAGER_DEPT_NAME
   FROM EMP E
   JOIN EMP M
   ON E.MGR = M.EMPNO
   JOIN DEPT D
   ON D.DEPTNO = E.DEPTNO
   JOIN DEPT DD
   ON DD.DEPTNO = M.DEPTNO;

2. SELECT E.ENAME, D.LOC AS EMP_LOCATION, M.ENAME AS MANAGER_NAME, DD.LOC AS MANAGER_LOCATION
   FROM EMP E
   JOIN EMP M
   ON E.MGR = M.EMPNO
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO
   JOIN DEPT DD
   ON DD.DEPTNO = M.DEPTNO
   WHERE D.LOC != DD.LOC;   

3. SELECT E.ENAME, E.SAL
   FROM EMP E
   WHERE E.SAL > (SELECT AVG(EE.SAL)
		  FROM EMP EE
		  JOIN DEPT D
		  ON EE.DEPTNO = D.DEPTNO
                  WHERE D.DNAME = 'SALES');

4. SELECT DISTINCT e.ename, e.job
   FROM emp e
   WHERE e.job IN (SELECT job FROM emp WHERE DEPTNO = 20);

5. SELECT D.DNAME, E.ENAME, E.SAL AS HIGHEST_SAL
   FROM DEPT D
   JOIN EMP E
   ON D.DEPTNO = E.DEPTNO
   WHERE E.SAL = (SELECT MAX(SAL) FROM EMP);

6. SELECT * FROM EMP 
   WHERE SAL > ( SELECT MAX(E.SAL) 
		 FROM EMP E 
		 JOIN 
		 DEPT D 
		 ON E.DEPTNO=D.DEPTNO  
		 WHERE D.DNAME='ACCOUNTING' ) ;

7. SELECT D.DNAME
   FROM DEPT D
   WHERE D.DEPTNO NOT IN (SELECT DEPTNO FROM EMP WHERE JOB = 'MANAGER');

8. SELECT E.ENAME, D.DNAME
   FROM EMP E
   JOIN DEPT D
   ON E.DEPTNO = D.DEPTNO
   WHERE E.DEPTNO NOT IN (SELECT DEPTNO FROM EMP WHERE JOB='CLERK');

9. SELECT ENAME, SAL, DEPTNO 
   FROM EMP 
   WHERE SAL > ALL(SELECT MAX(SAL) FROM EMP WHERE DEPTNO=30);

10. SELECT E.ENAME, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    WHERE D.DEPTNO IN (SELECT DEPTNO FROM EMP GROUP BY DEPTNO HAVING COUNT(*)>3);

11. SELECT E.ENAME, E.SAL
    FROM EMP E
    WHERE D.LOC='DALLAS' AND E.SAL > (SELECT AVG(SAL) FROM EMP E2 JOIN DEPT D ON E2.DEPTNO=D.DEPTNO );

12. SELECT D.DNAME, SUM(E.SAL) AS TOTAL_SAL
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    GROUP BY D.DNAME
    HAVING SUM(E.SAL) > 9000;

13. SELECT E.ENAME, D.DNAME
    FROM EMP E
    RIGHT JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO
    WHERE D.LOC='BOSTON';

14. SELECT E.ENAME, D.DNAME
    FROM EMP E
    JOIN DEPT D
    ON E.DEPTNO = D.DEPTNO
    WHERE E.HIREDATE > '31-DEC-1981';

15. SELECT E.ENAME, D.DNAME, EE.ENAME AS COLLEAGUE
    FROM EMP E
    JOIN EMP EE
    ON E.DEPTNO=EE.DEPTNO AND E.EMPNO <> EE.EMPNO
    JOIN DEPT D
    ON E.DEPTNO=D.DEPTNO;
